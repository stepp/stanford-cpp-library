43 /*
44  * Class: HashSet<ValueType>
45  * -------------------------
46  * This class implements an efficient abstraction for storing sets
47  * of distinct elements. This class is identical to the <code>Set</code>
48  * class except for the fact that it uses a hash table as its underlying
49  * representation. The advantage of the <code>HashSet</code> class is that
50  * it operates in constant time, as opposed to the <i>O</i>(log <i>N</i>)
51  * time for the <code>Set</code> class. The disadvantage of
52  * <code>HashSet</code> is that iterators return the values in a
53  * seemingly random order.
54  */
55 template <typename ValueType>
