859 void Vector<ValueType>::ensureCapacity(int cap) {
860  if (cap >= 1 && capacity < cap) {
861  capacity = std::max(cap, capacity * 2);
862  ValueType* array = new ValueType[capacity];
863  if (elements) {
864  for (int i = 0; i < count; i++) {
865  array[i] = elements[i];
866  }
867  delete[] elements;
868  }
869  elements = array;
870  }

931  * Implementation notes: insert, remove, add
932  * -----------------------------------------
933  * These methods must shift the existing elements in the array to
934  * make room for a new element or to close up the space left by a
935  * deleted one.

937 template <typename ValueType>
938 void Vector<ValueType>::insert(int index, const ValueType& value) {
939  checkIndex(index, 0, count, "insert");
940  if (count == capacity) {
941  expandCapacity();
942  }
943  for (int i = count; i > index; i--) {
944  elements[i] = elements[i - 1];
945  }
946  elements[index] = value;
947  count++;
948  m_version++;
949 }
