753  class GraphComparator {
754  public:
755  bool operator ()(NodeType* n1, NodeType* n2) {
756  return compare(n1, n2) < 0;
757  }
758 
759  bool operator ()(ArcType* a1, ArcType* a2) {
760  return compare(a1, a2) < 0;
761  }
762  };
763 
764 private:
765  Set<NodeType*> nodes; /* The set of nodes in the graph */
766  Set<ArcType*> arcs; /* The set of arcs in the graph */
767  Map<string, NodeType*> nodeMap; /* A map from names to nodes */
768  GraphComparator comparator; /* The comparator for this graph */
769 
770 public:
777  Graph& operator =(const Graph& src);
778 
785  Graph(const Graph& src);

792  static int compare(NodeType* n1, NodeType* n2) {
793  if (n1 == n2) {
794  return 0;
795  }
796  if (n1->name < n2->name) {
797  return -1;
798  }
799  if (n1->name > n2->name) {
800  return +1;
801  }
802  return (n1 < n2) ? -1 : +1;
803  }
804 
810  static int compare(ArcType* a1, ArcType* a2) {
811  if (a1 == a2) {
812  return 0;
813  }
814  NodeType* n1 = a1->start;
815  NodeType* n2 = a2->start;
816  if (n1 != n2) {
817  return compare(n1, n2);
818  }
819  n1 = a1->finish;
820  n2 = a2->finish;
821  if (n1 != n2) {
822  return compare(n1, n2);
823  }
824  return (a1 < a2) ? -1 : +1;
825  }

828  void deepCopy(const Graph& src);

914  * Implementation notes: addNode
915  * -----------------------------
916  * The addNode method appears in two forms: one that creates a node
917  * from its name and one that assumes that the client has created
918  * the new node. In each case, the implementation must add the node
919  * the set of nodes for the graph and add the name-to-node association
920  * to the node map.
921  */
922 template <typename NodeType, typename ArcType>
923 NodeType* Graph<NodeType, ArcType>::addNode(const string& name) {
924  NodeType* node = getNode(name);
925  if (node) {
926  return node; // vertex already exists
927  }
928  node = new NodeType();
929  node->arcs = Set<ArcType*>(comparator);
930  node->name = name;
931  return addNode(node);
932 }

1510 template <typename NodeType, typename ArcType>
1511 int Graph<NodeType, ArcType>::graphCompare(const Graph<NodeType, ArcType>& graph2) const {
1512  // optimization: if literally the same graph, return true
1513  if (this == &graph2) {
1514  return 0;
1515  }
1516 
1517  auto itr1 = begin();
1518  auto itr2 = graph2.begin();
1519  auto g1end = end();
1520  auto g2end = graph2.end();
1521 
1522  while (itr1 != g1end && itr2 != g2end) {
1523  // compare each pair of elements from iterators
1524  NodeType* node1 = *itr1;
1525  NodeType* node2 = *itr2;
1526 
1527  // optimization: if literally same node, equal; don't compare
1528  if (node1 != node2) {
1529  // first check names
1530  if (node1->name != node2->name) {
1531  return node1->name.compare(node2->name);
1532  }
1533 
1534  // then check all arcs, pairwise
1535  auto eitr1 = node1->arcs.begin();
1536  auto eitr2 = node2->arcs.begin();
1537  auto e1end = node1->arcs.end();
1538  auto e2end = node2->arcs.end();
1539  while (eitr1 != e1end && eitr2 != e2end) {
1540  ArcType* arc1 = *eitr1;
1541  ArcType* arc2 = *eitr2;
1542 
1543  // optimization: if literally same arc, equal; don't compare
1544  if (arc1 != arc2) {
1545  // first check start vertex names, then end vertex names
1546  if (arc1->start->name != arc2->start->name) {
1547  return arc1->start->name.compare(arc2->start->name);
1548  } else if (arc1->finish->name != arc2->finish->name) {
1549  return arc1->finish->name.compare(arc2->finish->name);
1550  }
1551  }
1552  eitr1++;
1553  eitr2++;
1554  }
1555 
1556  // if we get here, everything from me matched graph2, so either arcs equal,
1557  // or one is shorter than the other (fewer arcs) and is therefore less
1558  if (eitr1 == e1end && eitr2 == e2end) {
1559  // keep going
1560  } else if (eitr1 == e1end) {
1561  return -1;
1562  } else {
1563  return 1;
1564  }
1565  }
1566 
1567  // if we get here, those two vertices and their outbound arcs
1568  // were equal; so advance to next element
1569  itr1++;
1570  itr2++;
1571  }
1572 
1573  // if we get here, everything from me matched graph2, so either equal,
1574  // or one is shorter than the other (fewer vertices) and is therefore less
1575  if (itr1 == g1end && itr2 == g2end) {
1576  return 0;
1577  } else if (itr1 == g1end) {
1578  return -1;
1579  } else {
1580  return 1;
1581  }
1582 }
