65  * Performs a comparison for ordering between the given two collections
66  * by comparing their elements pairwise to each other.
67  * Returns -1 if collection 1 is "less than" collection 2;
68  * Returns 1 if collection 1 is "greater than" collection 2;
69  * Returns 0 if collection 1 is "equal to" collection 2.
70  * The element type must have an operator <.
71  */
72 template <typename CollectionType>
73 int compare(const CollectionType& coll1, const CollectionType& coll2) {
74  // optimization: if they are the same object, then they are equal
75  if (&coll1 == &coll2) {
76  return 0;
77  }
78 
79  auto itr1 = coll1.begin(),
80  itr2 = coll2.begin(),
81  end1 = coll1.end(),
82  end2 = coll2.end();
83  for (;
84  itr1 != end1 && itr2 != end2;
85  ++itr1, ++itr2) {
86  // compare each pair of elements from iterators
87 
88  // TO STUDENT:
89  // If the line below is failing to compile in your program, it probably
90  // means that you are trying to make a nested collection
91  // (e.g. Set<Vector<T>>) for some element type T that does not have a
92  // less-than < operator. That operator is *required* in order to make
93  // a Set or Map of Vectors, so that the set/map knows how to sort the
94  // elements into their ascending order.
95  // You should either add a < operator to your class, or consider a
96  // different nested collection solution. Good luck!
97  if (*itr1 < *itr2) {
98  return -1;
99  } else if (*itr2 < *itr1) {
100  return 1;
101  }
102  }
103 
104  // if we get here, everything from v1 matched v2, so they are either equal,
105  // or one is shorter than the other (fewer elements) and is therefore less
106  if (itr1 == end1 && itr2 == end2) {
107  return 0;
108  } else if (itr1 == end1) {
109  return -1;
110  } else {

116  * Performs a comparison for ordering between the given two maps
117  * by comparing their key/value pairs pairwise to each other.
118  * Returns -1 if collection 1 is "less than" collection 2;
119  * Returns 1 if collection 1 is "greater than" collection 2;
120  * Returns 0 if collection 1 is "equal to" collection 2.
121  * Note that the order of keys and values in the maps matter here;
122  * the same map with keys in different orders would produce a different result.
123  * The key and value types must both have an operator <.
124  */
125 template <typename MapType>
126 int compareMaps(const MapType& map1, const MapType& map2) {
127  // optimization: if they are the same object, then they are equal
128  if (&map1 == &map2) {
129  return 0;
130  }
131 
132  auto itr1 = map1.begin(),
133  itr2 = map2.begin(),
134  end1 = map1.end(),
135  end2 = map2.end();
136  for (;
137  itr1 != end1 && itr2 != end2;
138  ++itr1, ++itr2) {
139  // compare each pair of elements from iterators
140 
141  // TO STUDENT:
142  // If the line below is failing to compile in your program, it probably
143  // means that you are trying to make a nested collection
144  // (e.g. Set<Map<K, V>>) for some element type K or V that does not have a
145  // less-than < operator. That operator is *required* in order to make
146  // a Set or Map of Maps, so that the set/map knows how to sort the
147  // keys and values into their ascending order.
148  // You should either add a < operator to your key/value types, or consider a
149  // different nested collection solution. Good luck!
150  if (*itr1 < *itr2) {
151  return -1;
152  } else if (*itr2 < *itr1) {

157  auto value1 = map1[*itr1];
158  auto value2 = map2[*itr2];

161  } else if (value2 < value1) {

167  // or one is shorter than the other (fewer elements) and is therefore less

169  return 0;

287  * Returns true if the two collections contain the same elements in the same order.
288  * The element type must have an operator ==.
289  */
290 template <typename CollectionType>
291 bool equals(const CollectionType& coll1, const CollectionType& coll2) {
292  // optimization: if literally same collection, stop
293  if (&coll1 == &coll2) {
294  return true;
295  }
296  // optimization: if not same size, don't bother comparing pairwise
297  if (coll1.size() != coll2.size()) {
298  return false;
299  }
300 
301  // check each pair of elements for equality
302  auto itr1 = coll1.begin();
303  auto end1 = coll1.end();
304  auto itr2 = coll2.begin();
305  auto end2 = coll1.end();
306  while (itr1 != end1 && itr2 != end2) {
307  if (!(*itr1 == *itr2)) {
308  return false;
309  }
310  ++itr1;
311  ++itr2;
312  }
313  return true;
314 }

346 /*
347  * Returns true if the given two maps contain the same set of keys and each
348  * key maps to the same value in both maps.
349  * The order of the keys is not considered.
350  * The key and value types must both have an operator ==.
351  */
352 template <typename MapType>
353 bool equalsMap(const MapType& map1, const MapType& map2) {
354  // optimization: if literally same map, stop
355  if (&map1 == &map2) {
356  return true;
357  }
358  // optimization: if not same size, don't bother comparing pairwise
359  if (map1.size() != map2.size()) {
360  return false;
361  }
362 
363  // check each pair of key/value pairs for equality;
364  // compare both ways; each must be subset of the other
365  for (auto itr1 = map1.begin(), end1 = map1.end(); itr1 != end1; ++itr1) {
366  if (!map2.containsKey(*itr1) || !(map1.get(*itr1) == map2.get(*itr1))) {
367  return false;
368  }
369  }
370  for (auto itr2 = map2.begin(), end2 = map2.end(); itr2 != end2; ++itr2) {
371  if (!map1.containsKey(*itr2) || !(map1.get(*itr2) == map2.get(*itr2))) {
372  return false;
373  }
374  }
375  return true;
376 }

378 /*
379  * Computes a hashCode for any iterable type.
380  * Pass true for 'orderMatters' for collections that maintain some kind of element
381  * ordering that is of importance. This would be true for almost all collections
382  * except hash-based ones where the ordering is semi-random and unimportant.
383  */
384 template <typename IteratorType>
385 int hashCodeIterable(IteratorType begin, IteratorType end, bool orderMatters = true) {
386  int code = hashSeed();
387  while (begin != end) {
388  if (orderMatters) {
389  code *= hashMultiplier();
390  }
391  code += hashCode(*begin);
392  ++begin;
393  }
394  return int(code & hashMask());
395 }

406 /*
407  * Template hash function for hash maps.
408  * Requires the key and value types in the HashMap to have a hashCode function.
409  */
410 template <typename MapType>
411 int hashCodeMap(const MapType& map, bool orderMatters = true) {
412  int code = hashSeed();
413  auto begin = map.begin();
414  auto end = map.end();
415  while (begin != end) {
416  if (orderMatters) {
417  code *= hashMultiplier();
418  }
419  code += hashCode(*begin);
420 
421  if (orderMatters) {
422  code *= hashMultiplier();
423  }
424  code += hashCode(map[*begin]);
425  ++begin;
426  }
427  return int(code & hashMask());
428 }

472 #endif
